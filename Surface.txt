function createSurface(vertices, color){
	
	//Determine average distances squared (important for painting) and keep as depth
	var depth = 0;
	for (var i = 0; i < vertices.length; i++){
		depth += vertices[1] / 4; //depth here is determined by y coordinate
	}
	
	var surface = {
		vertices: vertices,
		color: color,
		depth: depth
	}
	
	return surface;
	
}
//--------------------------------------------------------------------------
function paintSurface(ctx, surface){
	
	var c = document.getElementById("graph");
	ctx.fillStyle = "#FF0000";
	ctx.beginPath();
	var coordinates = getXYProjection(surface.vertices[0], 3, Math.PI / 2, Math.PI / 2);
	ctx.moveTo(coordinates[0] * c.width, coordinates[1] * c.height);
	for (var i = 0; i < surface.vertices.length; i++){
		var coordinates2 = getXYProjection(surface.vertices[i], 3, Math.PI / 2, Math.PI / 2);
		ctx.lineTo(coordinates2[0] * c.width, coordinates2[1] * c.height);
	}
	ctx.lineTo(coordinates[0] * c.width, coordinates[1] * c.height);
	ctx.closePath();
	ctx.fill();
}
//--------------------------------------------------------------------------
function getLightFactor(surface){
	
	var lightVector = [0, 1, 0];
	
	var surfaceVector1 = sub(surface.vertices[1], surface.vertices[0]);
	var surfaceVector2 = sub(surface.vertices[2], surface.vertices[0]);
	var normalVector = unit(CP(surfaceVector1, surfaceVector2));
	return DP(normalVector, lightVector);
	
}
